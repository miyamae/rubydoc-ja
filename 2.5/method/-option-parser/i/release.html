instance method OptionParser#release

release -> String[permalink][rdoc]

プログラムのリリースを文字列で返します。 

例

require "optparse"

OptionParser.new do |opts|
  opts.release # => nil
  opts.release = "2019-05-01"
  opts.release # => "2019-05-01"
end
