instance method File::Stat#setgid?

setgid? -> bool[permalink][rdoc]

setgidされている時に真を返します。 


Dir.glob("/usr/sbin/*") {|bd|
  if File::Stat.new(bd).setgid?
    puts bd
  end
}
#例
#...
#=> /usr/sbin/postqueue
