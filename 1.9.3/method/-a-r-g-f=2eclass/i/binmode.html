require 'md5'

ARGV.replace(
  Dir.glob("*").select{|name| FileTest.file?(name)}
)
files = ARGV.dup
bm = ARGF.binmode.readlines(nil)
bm.each_with_index {|bl, i|
  md5 = MD5.new(bl).hexdigest
  printf("%s\t%s\n", md5, files[i])
}
