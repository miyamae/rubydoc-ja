instance method String#each_byte

each_byte {|byte| ... } -> self[permalink][rdoc]
each_byte -> Enumerator

文字列の各バイトに対して繰り返します。 

例

"str".each_byte do |byte|
  p byte
end
    # => 115
    # => 116
    # => 114

"あ".each_byte do |byte|
  p byte
end
    # => 227
    # => 129
    # => 130
