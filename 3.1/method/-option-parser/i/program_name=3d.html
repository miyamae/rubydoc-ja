instance method OptionParser#program_name=

program_name=(name)[permalink][rdoc]

プログラムの名前を文字列で指定します。

[PARAM] name:
プログラムの名前を文字列で指定します。 
例

require "optparse"

OptionParser.new do |opts|
  $0                         # => /path/to/filename.rb
  opts.program_name          # => filename
  opts.program_name = 'test' # => "test"
  opts.program_name          # => "test"
end
