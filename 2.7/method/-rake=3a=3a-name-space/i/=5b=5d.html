instance method Rake::NameSpace#[]

self[name] -> Rake::Task[permalink][rdoc]

与えられた名前のタスクを返します。

[PARAM] name:
タスクの名前を指定します。

namespace :ns do |ns|
  task :ts1 do
  end
  task :ts2 do
  end

  ns[:ts1] # => <Rake::Task ns:ts1 => []>
  ns[:ts2] # => <Rake::Task ns:ts2 => []>
end
