instance method Object#instance_variables

instance_variables -> [String]

オブジェクトのインスタンス変数名を文字列の配列として返します。 

obj = Object.new
obj.instance_eval { @foo, @bar = nil }
p obj.instance_variables

#=> ["@foo", "@bar"]
